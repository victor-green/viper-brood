### LOOPS
```python


value = 100


# while
while value == 100:
	print "value is 100"
	value += 1


# while else
while value == 100
    print "value is 100"
    value += 1
else:
    print "value is not 100"


# for
fruits = ['apple','pear','plum']
for fruit in fruits:
	print fruit


# for else
fruits = ['apple','pear','plum']
for fruit in fruits:
	print fruit
else:
    print "List complete"


# for iteration by index
for index in range(len(fruits))
	print fruits[index]


# break
while True:
	value += 1
	if value == 130:
	   break


# continue
while True:
    value += 1
    if value == 110:
       continue:
    elif value == 120:
       break



```
